(*本程序用于计算投加流量*)
if auto_manual_switch then
    (*自动控制时，投加流量由计算公式确定*)
    (*计算公式：*)
    (*1m3=1000L;1mg=1/(1000*1000)kg*)
    (*用量(kg/h)=药耗(mg/L)*处理水量(m3/h)/(百分比浓度%*1000*)
    (*投加流量(L/h)=药剂密度(kg/m3)*用量(kg/h)*)
    quantity := dosing_rate_mg_L * source_water_flow / (concentration*1000.0);
    flow_setting := density * quantity;
else
    (*手动控制时，投加流量由触摸屏控制*)
    flow_setting := manual_flow_setting;
end_if;
